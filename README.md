# Công Thức Đồng Dư :

  - (A + B) % C = ((A % C) + (B % C)) % C
  - (A - B) % C = ((A % C) - (B % C)) % C
  - (A * B) % C = ((A % C) * (B % C)) % C
  - (A / B) % C = ((A % C) * (B-1 % C)) % C
